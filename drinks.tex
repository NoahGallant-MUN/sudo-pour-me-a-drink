\part{Aperol}
\begin{drink}{Candy Trail}
	\begin{ingredients}
		\item 6 basil leaves
		\item \threequarter\ ounce lemon juice
		\item \half\ ounce apricot brandy
		\item \half\ ounce peach liqueur
		\item 1 ounce Aperol
	\end{ingredients}
	\begin{instructions}
		Add all liquids to a cocktail shaker filled with ice. Tear and rub together the basil leaves in your hand to bring out the flavour. Add the leaves to the shaker and shake vigorously. Double strain into a chilled coupe glass. Garnish with one or more basil leaves.
	\end{instructions}
\end{drink}

\part{Gin}

\begin{drink}{Basil Smash}
	\begin{ingredients}
		\item 2 ounces Gin
		\item \threequarter\ ounce Fresh Lemon Juice
		\item \threequarter\ ounce Simple Syrup
		\item 4--6 Leaves of Fresh Basil
	\end{ingredients}
	\begin{instructions}
		Add two ounces of gin to a cocktail shaker, followed by three quarters of an ounce of both fresh lemon juice and simple syrup, and finally four to six leaves (torn in half) of fresh basil. When you shake the cocktail, the ice will bruise the basil, imparting its herbal notes on the drink. Fill the cocktail shaker with ice, shake vigorously, and strain over ice into an old fashioned glass. Garnish with a sprig of basil.		
	\end{instructions}
\end{drink}

\begin{drink}{Bramble}
	\begin{ingredients}
	\item 1-\half\ ounces Gin
	\item \threequarter\ ounce Fresh Lemon Juice
	\item \half\ ounce Simple Syrup
	\item \threequarter\ ounce Cr\`eme de Mure
	\end{ingredients}
	\begin{instructions}
		This cocktail is build in the glass. Fill half the glass with crushed ice and add one and a half ounces of your preffered gin, three quarters of an ounce of fresh lemon juice, and half an ounce of simple syrup. Overload the glass with ice and add three quarters of an ounce of Cr\`eme de Mure over the top of the ice to cascade down the sides of the glass. Garnish with raspberries or blackberries.
	\end{instructions}
\end{drink}

\begin{drink}{Clover Club}
	\begin{ingredients}
		\item 1-\half\ ounces Gin
		\item \threequarter\ ounce Fresh Lemon Juice
		\item \threequarter\ ounce Raspberry Syrup
		\item Fresh Egg White
	\end{ingredients}
	\begin{instructions}
		Add one and a half ounces of gin and three quarters of an ounce of both lemon juice and raspberry syrup to a mixing glass. Add an egg white to the glass and dry shake; this will emulsify the egg. Add ice to the glass and shake again. Finish by straining into a chilled coupe glass. This drink does not traditionally have a garnish but you may add one.
	\end{instructions}
\end{drink}

\begin{drink}{Corpse Reviver N\textsuperscript{\d o} 2}
	\begin{ingredients}
		\item 1 ounce Gin
		\item 1 ounce Cointreau orange liqueur 
		\item 1 ounce Lillet Blanc\footnote{\em Substitutable with dry vermouth.}
		\item 1 ounce fresh lemon juice
		\item Dash of Absinthe
	\end{ingredients}
	\begin{instructions}
		To a cocktail shaker, add an ounce of gin, Cointreau, Lillet Blanc, and fresh lemon juice. Add a slash of absinthe and fill the shaker with ice. Shake well and strain into a chilled stemmed cocktail glass. Garnish with a twist of lemon, twisting to squeeze some of the oils over the drink.
	\end{instructions}
\end{drink}

\begin{drink}{Green Eyes}
	\begin{ingredients}
			\item \half\ ounce egg white
			\item 1-\half\ ounces gin
			\item \threequarter\ ounce Green Chartreuse
			\item \threequarter\ ounce fresh lime juice
			\item \half\ ounce rich (2:1) simple syrup
	\end{ingredients}
	\begin{instructions}
		Add half an ounce of egg white to your cocktail shaker, followed by a single ice cube, and dry shake. Next, add an ounce and a half of gin and three quarters of an ounce of both Green Chartreuse and fresh lime juice. To finish, add half an ounce of rich simple syrup, fill with ice, shake, and serve in a chilled double old fashioned glass.
	\end{instructions}
\end{drink}

\begin{drink}{Green Ghost}
	\begin{ingredients}
		\item 2 ounces gin
		\item \half\ ounce Green Chartreuse
		\item \half\ ounce fresh lime juice
	\end{ingredients}
	\begin{instructions}
		Fill a cocktail shaker with ice. Add two ounces of gin and half an ounce of both Green Chartreuse and fresh lime juice. Shake and strain into a martini glass.
	\end{instructions}
\end{drink}

\begin{drink}{The Last Word}
	\begin{ingredients}
		\item \threequarter\ ounce Gin
		\item \threequarter\ ounce Green Chartreuse
		\item \threequarter\ ounce Maraschino Liqueur 
		\item \threequarter\ ounce Fresh Lime Juice
	\end{ingredients}
	\begin{instructions}
		Starting with a shaker, add equal parts --- three quarters of an ounce --- of gin, green Chartreuse, Maraschino liqueur, and fresh lime juice. Fill your cocktail shaker with ice and give it a good shake. Double strain into a chilled coupe glass. Though often left ungarnished, a lime twist is an appropriate garnish.
	\end{instructions}
\end{drink}

\begin{drink}{Negroni}
	\begin{ingredients}
		\item \half\ ounce Campari
		\item \half\ ounce sweet vermouth
		\item 1-\half\ ounces gin
	\end{ingredients}
	\begin{instructions}
		This cocktail is built directly in the glass. Place a large sphere or cube of ice in an old fashioned glass and pour in half an ounce of both Campari and sweet vermouth, followed by an ounce and a half of gin. Stir a little to incorporate and reach the desired dilution. Garnish with the zest of a lemon.
	\end{instructions}
\end{drink}

\begin{drink}{The Sky is the Lemon}
	\begin{ingredients}
		\item 1 ounce Lab Infernal Ginger syrup
		\item \threequarter\ ounce fresh lemon juice
		\item \half\ ounce Galliano vanilla liqueur
		\item \half\ ounce Massenez lime and ginger liqueur
		\item 1 ounce gin
		\item 35\% cream, whipped
		\item Vanilla sugar
	\end{ingredients}
	\begin{instructions}
		Whip the cream until thick but unable to form light peaks and set aside. Pour all ingredients (except cream and sugar) into a cocktail shaker filled with ice. Shake vigorously. Filter into a martini glass. Gently pour the partially-whipped cream on the top of the cocktail. Sprinkle with vanilla sugar. Garnish with grated lime peel.
	\end{instructions}
\end{drink}

\begin{drink}{Southside}
	\begin{ingredients}
		\item 2 ounces gin
		\item \threequarter\ ounce fresh lime juice
		\item \threequarter\ ounce simple syrup
		\item 4--5 mint leaves
	\end{ingredients}
	\begin{instructions}
		Fill a Boston glass with ice. Add two ounces of gin, followed by three quarters of an ounce of both fresh lime juice and simple syrup. Finish by adding four or five mint leaves to your glass and shake. Double strain into a chilled coupe glass.
	\end{instructions}
\end{drink}

\begin{drink}{No-'Merica}
	\begin{ingredients}
		\item 3 dashes of cherry bitters
		\item 1 ounce Aperol
		\item 1 ounce sweet vermouth
		\item Elderflower tonic water
	\end{ingredients}
	\begin{instructions}
		Build this cocktail in an old fashioned glass. Fill the glass with ice and add three dashes of cherry bitters, an ounce of Aperol, an ounce of vermouth, and then the tonic. Stir gently with a bar spoon. Garnish with a slice of lemon. 
	\end{instructions}
\end{drink}

\part{Vodka}

\begin{drink}{Cosmolab}
	\begin{ingredients}
		\item 1-\half\ ounces vodka
		\item \half\ ounce Pierre Ferrand Dry Cura\c cao
		\item \onequarter\ lime, pressed
		\item \onequarter\ ounce Lab Infernal Ginger syrup
		\item 1-\half\ ounces cranberry juice
	\end{ingredients}
	\begin{instructions}
		Place all ingredients in a shaker and add ice. Shake well and pour with a strainer into a martini glass. Sprinkle with golg and garnish with a cherry.
	\end{instructions}
\end{drink}

\begin{drink}{Caipiroska}
	\begin{ingredients}	
		\item 2 ounces vodka
		\item \half\ to \threequarter\ of a lime
		\item 3 bar spoons sugar
	\end{ingredients}
	\begin{instructions}
		Place four to six eighths of a lime into a cocktail shaker and add three bar spoons of sugar, then muddle the ingredients in the shaker. Fill the cocktail shaker with ice and add two ounces of vodka. Shake and pour into a chilled old fashioned glass.
	\end{instructions}
\end{drink}

\begin{drink}{Harvey Wallbanger}
	\begin{ingredients}
		\item \onequarter\ ounce Galliano
		\item 2 ounces vodka
		\item 4 ounces fresh orange juice
	\end{ingredients}
	\begin{instructions}
		Build this cocktail in a highball or rocks glass by adding one quarter of an ounce of Galliano liqueur, followed by two ounces of vodka. Fill your glass with ice and pour in four ounces of fresh orange juice. Stir and serve.
	\end{instructions}
\end{drink}

\begin{drink}{Sex on the Beach}
	\begin{ingredients}
		\item 1-\half\ ounces Vodka
		\item \half\ ounce Peach Liqueur
		\item 1-\half\ ounces Fresh Orange Juice
		\item 1-\half\ ounces Fresh Cranberry Juice
	\end{ingredients}
	\begin{instructions}
		Start with a hurricane glass. Fill the glass with ice and add an ounce and a half of vodka, followed by half an ounce of peach liqueur. Next add one and a half ounces of fresh orange juice. Top up the glass with ice and {\em gently} add an ounce and a half of fresh cranberry juice. Garnish with a slice of orange and a maraschino cherry on a cocktail pick.
	\end{instructions}
	
\end{drink}

\part{Rum}

\begin{drink}{Long Island Iced Tea}
	\begin{ingredients}
		\item \half\ ounce White Rum
		\item \half\ ounce Vodka
		\item \half\ ounce Gin
		\item \half\ ounce Tequila 
		\item \half\ ounce Cointreau
		\item Coke
		\item \threequarter\ ounce Fresh Lemon Juice
	\end{ingredients}
	\begin{instructions}
		Build this cocktail in a highball glass. Fill the glass half way with ice and add half an ounce of white rum, vodka, gin, tequila, and Cointreau. Top the glass up with ice and fill almost to the top with coke. Don't add too much coke; you should still be able to see through the drink. Add three quarters of an ounce of lemon juice and then garnish with a lime wedge.
	\end{instructions}
\end{drink}

\begin{drink}{The President}
	\begin{ingredients}
		\item 1-\half\ ounces amber rum
		\item \half\ ounce dry vermouth
		\item 1 tablespoon Cointreau orange liqueur
		\item 1 bar spoon vintage Grenadine
	\end{ingredients}
	\begin{instructions}
		Add ice to a mixing glass, followed by all of the ingredients. Stir with a bar spoon to cool and reach the desired dilution. Strain into a chilled coupe glass. Garnish with a candied cherry or orange zest.
	\end{instructions}
\end{drink}

\begin{drink}{Rastafire}
	\begin{ingredients}
		\item 2 bar spoons of maple syrup
		\item 2 drops Jamaican Jerk bitters
		\item 1 dash chocolate bitters
		\item \half\ ounce peated single malt scotch
		\item 1-\half\ ounces dark rum
	\end{ingredients}
	\begin{instructions}
		Add ice to a mixing glass then add two bar spoons of maple syrup, two drops of Jamaican Jerk bitters, a dash of chocolate bitters, half an ounce of a peaty single malt scotch, and one and a half ounces of dark rum. Stir gently with a bar spoon to mix. Pour by filtering into an old fashioned glass containing a large ball or cube of ice. Garnish with an orange zest.
	\end{instructions}
\end{drink}

		\item \half\ ounce Campari
		\item \half\ ounce Orgeat
		\item \half\ ounce Fresh Lime Juice
		\item \threequarter\ ounce Strawberry Syrup

\part{Whiskey}

\begin{drink}{Blood \& Sand}
	\begin{ingredients}
		\item \threequarter\ ounce Scotch Whiskey%
			\footnote{Smokey whiskies may be overpowering in this drink. For this reason, avoid Islay whiskies here; steer towards a blended whiskey.}
		\item \threequarter\ ounce Sweet Vermouth
		\item \threequarter\ ounce Heering Cherry Liqueur
		\item \threequarter\ once Fresh Orange Juice
	\end{ingredients}
	\begin{instructions}
		This cocktail is a shaken cocktail so add three quarters of an ounce of Scotch whiskey to your cocktail shaker, followed by equal measures of sweet vermouth, cherry liqueur, and fresh orange juice. Fill your shaker with ice and shake vigorously. Strain into a chilled coupe glass and add a twist of orange as a garnish. 
	\end{instructions}
\end{drink}

\begin{drink}{Bobby Burns}
	\begin{ingredients}
		\item 1-\half\ ounces Blended Scotch Whiskey
		\item 1-\half\ ounces Sweet Vermouth
		\item \threequarter\ ounce Dom Benedictine
	\end{ingredients}
	\begin{instructions}
		Build this cocktail in a mixing glass by adding an ounce and a half of blended Scotch whiskey, followed by one and half ounces of sweet vermouth and finally three quarters of an ounce of Dom Benedictine. Fill the mixing glass with ice and stir for about thirty seconds. Strain into a chilled Nick and Nora glass. To garnish add a twist of lemon, bending or twisting it over the drink to express the oils in the lemon.
	\end{instructions}
\end{drink}

\begin{drink}{Rob Roy}
	\footnotetext{The Rob Roy is essentially a Scotch Manhattan. It will be a slightly less sweet version of a classic Manhattan.}
	\begin{ingredients}

		\item \threequarter\ ounce Sweet Vermouth
		\item 2 dashes Angostura Bitters
	\end{ingredients}
	\begin{instructions}
		Add to your mixing glass one and a half ounces of your preferred Scotch whiskey, followed by three quarters of an ounce of sweet vermouth and half an ounce of Angostura bitters. Fill the mixing glass with ice and stir for about thirty seconds. Strain into a chilled coupe glass and garnish with a mara\-schino cherry.
	\end{instructions}
	
\end{drink}

\begin{drink}{Tipperary}
	\begin{ingredients}
		\item 1-\half ounces (or 1 ounce) Irish Whiskey
		\item 1 ounce Sweet Vermouth
		\item \half ounce (or 1 ounce) Green Chartreuse
	\end{ingredients}
	\begin{instructions}
		To a mixing glass add an ounce and a half of whiskey, an ounce of sweet vermouth, and half an ounce of Green Chartreuse. If you are a fan of Chartreuse, feel free to use equal parts (one ounce) of all ingredients instead. Fill the mixing glass with ice and stir using a bar spoon for thirty to fourty seconds. Strain into a chilled coupe glass and garnish with a maraschino cherry on a cocktail pick.
	\end{instructions}
\end{drink}
